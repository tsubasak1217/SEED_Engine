// オブジェクトの種類
enum class StageObjectType {

    None = 0,          // 空白
    NormalBlock = 1,   // 通常ブロック
    Goal = 2,          // ゴール
    Player = 3,        // プレイヤー
    Warp = 4,          // ワープ
    EmptyBlock = 5,    // 非表示ブロック
    LaserLauncher = 6, // レーザー発射台
};
// 発射台について
方向はU(UP)D(DOWN)L(LEFT)R(RIGHT)を横につけてください
順番は何でも大丈夫です
例.6R,6RB,6DL,6TBLR
